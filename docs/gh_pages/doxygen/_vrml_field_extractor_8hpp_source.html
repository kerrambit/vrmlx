188 inline cpp::result<std::reference_wrapper<const vrml_proc::parser::model::VrmlNode>, ExtractVrmlNodeError>
189 ExtractVrmlNodeExtended(const std::string& name,
190 const std::vector<vrml_proc::parser::model::VrmlField>& fields,
191 const vrml_proc::parser::service::VrmlNodeManager& manager,
192 std::string& invalidType,
193 std::string& useId) { //
194
195 if (IsNamePresent(name, fields)) {
196 std::string error = "";
197 auto vrmlNode = ExtractByNameExtended<vrml_proc::parser::model::VrmlNode>(name, fields, error);
198 if (vrmlNode.has_value()) {
199 return vrmlNode.value();
200 }
201
202 auto useNode = ExtractByName<vrml_proc::parser::model::UseNode>(name, fields);
203 if (useNode.has_value()) {
204 auto managerFound = manager.GetDefinitionNode(useNode.value().get().identifier);
205 if (managerFound.has_value()) {
206 return managerFound.value();
207 } else {
208 useId = useNode.value().get().identifier;
209 return cpp::fail(ExtractVrmlNodeError::UnknownUseNode);
210 }
211 }
212
213 invalidType = error;
214 return cpp::fail(ExtractVrmlNodeError::ValidationError);
215 }
216
217 return cpp::fail(ExtractVrmlNodeError::FieldNotFound);
218 }

519 cpp::result<std::reference_wrapper<const T>, std::optional<std::string>> operator()(const float& value) const {
520 using namespace vrml_proc::core::logger;
521 using namespace vrml_proc::core::utils;
522 using namespace vrml_proc::parser::model::utils;
523
524 LogDebug(FormatString("Visit float32_t. Object address is <", &value, ">."), LOGGING_INFO);
525
526 if constexpr (std::is_same<T, float>::value) {
527 LogDebug("Extract as float32_t.", LOGGING_INFO);
528 return std::cref(value);
529 }
530
531 if constexpr (std::is_same<T, int32_t>::value) {
532 if (value == static_cast<int32_t>(value)) {
533 static ExtractorCache cache;
534 auto intValue = cache.GetInt(reinterpret_cast<std::uintptr_t>(&value));
535
536 if (intValue.has_value()) {
537 LogDebug(FormatString(
538 "Edge case: float32_t is a whole number and thus can be extracted as int32_t with address: <",
539 &(intValue.value()), ">."),
540 LOGGING_INFO);
541 return intValue.value();
542 } else {
543 cache.StoreInt(reinterpret_cast<std::uintptr_t>(&value), static_cast<int32_t>(value));
544 auto newIntValue = cache.GetInt(reinterpret_cast<std::uintptr_t>(&value)).value();
545 LogDebug(FormatString(
546 "Edge case: float32_t is a whole number and thus can be extracted as int32_t with address: <",
547 &(newIntValue), ">."),
548 LOGGING_INFO);
549 return newIntValue;
550 }
551 }
552 }
553
554 LogDebug("Float32_t could not be extracted.", LOGGING_INFO);
555 return cpp::fail(std::optional<std::string>(TypeToString<float>()));
556 }
