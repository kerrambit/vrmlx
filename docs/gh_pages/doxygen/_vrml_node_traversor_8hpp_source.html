108 // Validate node using node descriptor.
109 auto& nodeDescriptor = nodeDescriptorResult.value();
110 auto validationResult = nodeDescriptor.Validate(params.node.get(), m_manager, m_headersMap, false);

112 LogError(FormatString("Validation for node <", canonicalHeader, "> failed!"), LOGGING_INFO);
113 return cpp::fail(std::make_shared<NodeTraversorError>(validationResult.error(), params.node.get()));
114 }
115
116 // Find and run handler for node.
117 auto handlerResult = FindAndRunHandler(canonicalHeader, params, validationResult.value());
118 if (handlerResult.has_error()) {
