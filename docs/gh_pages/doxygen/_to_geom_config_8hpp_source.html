137 parallelismSettings.active = parallelism.value("active", true);
138 parallelismSettings.threadsNumberLimit =
139 parallelism.value("threadsNumberLimit", std::thread::hardware_concurrency());
140 if (parallelismSettings.threadsNumberLimit <= 0) {
141 parallelismSettings.threadsNumberLimit = 1;
142 }

149 meshSimplificationSettings.percentageOfAllEdgesToSimplify =
150 vrml_proc::core::utils::UnitInterval::Create(percentOfAllEdgesToSimplify / 100).value();
