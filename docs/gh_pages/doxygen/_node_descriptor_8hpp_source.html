128 std::shared_ptr<vrml_proc::traversor::validation::error::NodeValidationError>>
129 Validate(const vrml_proc::parser::model::VrmlNode& node,
130 const vrml_proc::parser::service::VrmlNodeManager& manager,
131 const VrmlHeaders& headersMap,
132 bool checkName = false) { //
133
134 using namespace vrml_proc::traversor::validation::NodeValidationUtils;
135 using namespace vrml_proc::traversor::validation::error;
136
137 // ---------------------------------------------------
138
139 // First, we may want to check node's name.
140 if (checkName && ((node.header != m_name) && (headersMap.ConvertToCanonicalHeader(node.header) != m_name))) {
141 auto expectedHeaders = headersMap.GetSynonymsForCanonicalHeaders({node.header});
142 expectedHeaders.insert(m_name);
143 return cpp::fail(std::make_shared<InvalidVrmlNodeHeader>(node.header, expectedHeaders));
144 }
145
146 // We copy default values into possible future result NodeView.
147 NodeView::Builder builder;
148 builder.SetName(m_name);
149 builder.SetDefaultValues(m_fieldTypes, m_defaultBoolFields, m_defaultStringFields, m_defaultFloat32Fields,
150 m_defaultInt32Fields, m_defaultVec2fFields, m_defaultVec3fFields, m_defaultVec4fFields,
151 m_defaultVec2fArrayFields, m_defaultVec3fArrayFields, m_defaultInt32ArrayFields, m_defaultNodeFields,
152 m_defaultNodeArrayFields);
153
154 // Nothing to check here.
155 if (node.fields.empty()) {
156 return builder.Build();
157 }
158
159 auto fieldsResult =
160 CheckForOnlyUniqueAllowedFieldNames(ExtractFieldNames(m_fieldTypes), node.fields, node.header);
161 if (fieldsResult.has_error()) {
162 return cpp::fail(fieldsResult.error());
163 }
164
165 // Iterate through all fields and checks types.
166 for (const auto& field : node.fields) {
167 FieldType type = m_fieldTypes[field.name];
168 switch (type) {
169 case FieldType::Node:
170
171 {
172 auto vrmlNode = ExtractVrmlNodeWithValidation(field.name, node.fields, manager);
173 if (vrmlNode.has_error()) {
174 return cpp::fail(vrmlNode.error());
175 }
176
177 if (vrmlNode.value().has_value()) {
178 auto headerResult = CheckForOnlyAllowedVrmlNodeHeaders(
179 headersMap.GetSynonymsForCanonicalHeaders(m_validHeaderNames[field.name]),
180 vrmlNode.value().value().get(), field.name);
181 if (headerResult.has_error()) {
182 return cpp::fail(headerResult.error());
183 }
184 }
185 builder.AddField(field.name, vrmlNode.value());
186 } break;
187
188 case FieldType::NodeArray:
189
190 {
191 auto vrmlNodeArray = ExtractVrmlNodeArrayWithValidation(field.name, node.fields, manager, true);
192 if (vrmlNodeArray.has_error()) {
193 return cpp::fail(vrmlNodeArray.error());
194 }
195 builder.AddField(field.name, vrmlNodeArray.value());
196 } break;
197
198 case FieldType::Vec3f:
199
200 {
201 auto vec3f = ExtractFieldByNameWithValidation<vrml_proc::parser::model::Vec3f>(field.name, node.fields);
202 if (vec3f.has_error()) {
203 return cpp::fail(vec3f.error());
204 }
205 builder.AddField(field.name, vec3f.value());
206 } break;
207
208 case FieldType::Vec3fArray:
209
210 {
211 auto value =
212 ExtractFieldByNameWithValidation<vrml_proc::parser::model::Vec3fArray>(field.name, node.fields);
213 if (value.has_error()) {
214 return cpp::fail(value.error());
215 }
216 builder.AddField(field.name, value.value());
217 } break;
218
219 case FieldType::Int32Array:
220
221 {
222 auto value =
223 ExtractFieldByNameWithValidation<vrml_proc::parser::model::Int32Array>(field.name, node.fields);
224 if (value.has_error()) {
225 return cpp::fail(value.error());
226 }
227 builder.AddField(field.name, value.value());
228 } break;
229
230 case FieldType::Float32:
231
232 {
233 auto float32_t =
234 ExtractFieldByNameWithValidation<vrml_proc::parser::model::float32_t>(field.name, node.fields);
235 if (float32_t.has_error()) {
236 return cpp::fail(float32_t.error());
237 }
238 builder.AddField(field.name, float32_t.value());
239 } break;
240
241 case FieldType::Int32:
242
243 {
244 auto int32 = ExtractFieldByNameWithValidation<int32_t>(field.name, node.fields);
245 if (int32.has_error()) {
246 return cpp::fail(int32.error());
247 }
248 builder.AddField(field.name, int32.value());
249 } break;
250
251 case FieldType::Vec2f:
252
253 {
254 auto vec2f = ExtractFieldByNameWithValidation<vrml_proc::parser::model::Vec2f>(field.name, node.fields);
255 if (vec2f.has_error()) {
256 return cpp::fail(vec2f.error());
257 }
258 builder.AddField(field.name, vec2f.value());
259 } break;
260
261 case FieldType::Vec4f:
262
263 {
264 auto vec4f = ExtractFieldByNameWithValidation<vrml_proc::parser::model::Vec4f>(field.name, node.fields);
265 if (vec4f.has_error()) {
266 return cpp::fail(vec4f.error());
267 }
268 builder.AddField(field.name, vec4f.value());
269 } break;
270
271 case FieldType::Vec2fArray:
272
273 {
274 auto value =
275 ExtractFieldByNameWithValidation<vrml_proc::parser::model::Vec2fArray>(field.name, node.fields);
276 if (value.has_error()) {
277 return cpp::fail(value.error());
278 }
279 builder.AddField(field.name, value.value());
280 } break;
281
282 case FieldType::Bool:
283
284 {
285 auto boolean = ExtractFieldByNameWithValidation<bool>(field.name, node.fields);
286 if (boolean.has_error()) {
287 return cpp::fail(boolean.error());
288 }
289 builder.AddField(field.name, boolean.value());
290 } break;
291
292 case FieldType::String:
293
294 {
295 auto string = ExtractFieldByNameWithValidation<std::string>(field.name, node.fields);
296 if (string.has_error()) {
297 return cpp::fail(string.error());
298 }
299 if (m_validEnumValues.find(field.name) != m_validEnumValues.end()) {
300 if (m_validEnumValues[field.name].find(string.value().value().get()) ==
301 m_validEnumValues[field.name].end()) {
302 return cpp::fail(std::make_shared<InvalidStringValueError>(
303 node.header, field.name, string.value().value().get(), m_validEnumValues[field.name]));
304 }
305 }
306 builder.AddField(field.name, string.value());
307 } break;
308
309 default:
310 break;
311 }
312 }
313
314 return builder.Build();
315 }
