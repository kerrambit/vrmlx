37 inline cpp::result<void, std::shared_ptr<error::NodeValidationError>> CheckForOnlyUniqueAllowedFieldNames(
38 const std::unordered_set<std::string>& validFieldNames,
39 const std::vector<vrml_proc::parser::model::VrmlField>& fields,
40 const std::string& nodeName) { //
41
42 std::unordered_set<std::string> alreadyFoundFieldNames;
43 for (const auto& field : fields) {
44 if (validFieldNames.find(field.name) == validFieldNames.end()) {
45 return cpp::fail(std::make_shared<error::InvalidVrmlFieldName>(nodeName, field.name, validFieldNames));
46 }
47
48 if (alreadyFoundFieldNames.find(field.name) != alreadyFoundFieldNames.end()) {
49 return cpp::fail(std::make_shared<error::DuplicatedVrmlFieldName>(field.name));
50 }
51 alreadyFoundFieldNames.insert(field.name);
52 }
53 return {};
54 }

177 using namespace vrml_proc::parser;

179
180 if (!IsNamePresent(fieldName, fields)) {
181 return std::optional<std::vector<std::reference_wrapper<const model::VrmlNode>>>{};
182 }
183
184 if (enableSingleArrayNode) {
185 std::string invalidType = "";
186 std::string missingUseId = "";
187 auto value = ExtractVrmlNodeExtended(fieldName, fields, manager, invalidType, missingUseId);

192 }
193 } else {
194 std::vector<std::reference_wrapper<const model::VrmlNode>> resolvedChildren;
195 resolvedChildren.push_back(value.value());

197 }
198 }
199
200 std::string invalidType = "";
201 auto value = ExtractByNameExtended<model::VrmlNodeArray>(fieldName, fields, invalidType);
202
203 if (value.has_error()) {
204 if (value.error() == ExtractByNameError::FieldNotFound) {
205 return std::optional<std::vector<std::reference_wrapper<const model::VrmlNode>>>{};
206 } else if (value.error() == ExtractByNameError::ValidationError) {
207 return cpp::fail(std::make_shared<error::InvalidFieldValueType>(
208 fieldName, TypeToString<model::VrmlNodeArray>(), invalidType));
209 }
210 }
211
212 std::vector<std::reference_wrapper<const model::VrmlNode>> resolvedChildren;
213
214 for (const auto& child : value.value().get()) {
215 std::string invalidType;
216 std::string useId;
217 auto childResult = ExtractVrmlNodeFromVariantExtended(child, manager, invalidType, useId);
218 if (childResult.has_error()) {
219 switch (childResult.error()) {
220 case ExtractVrmlNodeFromVariantError::ValidationError:
221 return cpp::fail(std::make_shared<error::InvalidFieldValueType>(
222 fieldName, TypeToString<vrml_proc::parser::model::VrmlNode>(), invalidType));
223 case ExtractVrmlNodeFromVariantError::UnknownUseNode:
224 return cpp::fail(std::make_shared<error::MissingDefNodeForUseNode>(useId));
225 default:
226 break;
227 }
228 }
229 resolvedChildren.push_back(childResult.value());
230 }
231
232 return resolvedChildren;
